最小费用流

//CodeForces 237E
#include <iostream>
#include <cstring>
#include <cstdio>
#include <set>
#include <vector>
#include <algorithm>
using namespace std;

const int INF=1<<30;
const int MAXSIZE=300;

struct  Edge
{
	int u,v,w,c;
	int next;
}buf[MAXSIZE*MAXSIZE];

int head[MAXSIZE];
bool inq[MAXSIZE];
int pre[MAXSIZE];
int nE,ans;

void addEdge(int u,int v,int w,int c)
{
	buf[nE].u=u;buf[nE].v=v;buf[nE].w=w;buf[nE].c=c;buf[nE].next=head[u];head[u]=nE++;
	buf[nE].u=v;buf[nE].v=u;buf[nE].w=-w;buf[nE].c=0;buf[nE].next=head[v];head[v]=nE++;
}

int dis[MAXSIZE];
bool spfa(int src,int des)
{
	memset(inq,false,sizeof(inq));
	for(int i=0;i<=des;i++)
		dis[i]=INF;
	dis[src]=0;
	int frt=0,tail=0;
	int que[MAXSIZE];
	inq[src]=true;
	que[tail++]=src;
	pre[src]=-1;

	while(frt<tail)
	{
		int u=que[frt++];
		inq[u]=false;
		for(int i=head[u];i!=-1;i=buf[i].next)
		{
			int v=buf[i].v;
			if(dis[u]+buf[i].w<dis[v]&&buf[i].c>0)
			{
				pre[v]=i;
				dis[v]=dis[u]+buf[i].w;
				if(!inq[v])
				{
					inq[v]=true;
					que[tail++]=v;
				}
			}
		}
	}
	return dis[des]<INF;
}

int modify(int des)
{
	int mf=INF;
	for(int i=pre[des];i!=-1;i=pre[buf[i].u])
		if(buf[i].c<mf)
			mf=buf[i].c;
	for(int i=pre[des];i!=-1;i=pre[buf[i].u])
	{
		buf[i].c-=mf;
		buf[i^1].c+=mf;
		ans+=mf*buf[i].w;
	}
	return mf;
}

int main()
{
	memset(head,0xff,sizeof(head));
	char str[110];
	cin>>str;

	int ccnt[26]={0},need=0;
	for(int i=0;str[i];i++)
		ccnt[str[i]-'a']++;
	for(int i=0;i<26;i++)
		if(ccnt[i])
		{
			need+=ccnt[i];
			addEdge(0,i+1,0,ccnt[i]);
		}

	int n;
	cin>>n;
	int dst=n*2+26+1;

	for(int i=1,num;i<=n;i++)
	{
		cin>>str>>num;
		memset(ccnt,0,sizeof(ccnt));
		for(int j=0;str[j];j++)
			ccnt[str[j]-'a']++;
		for(int j=0;j<26;j++)
			if(ccnt[j])
				addEdge(j+1,i+26,0,ccnt[j]);
		addEdge(i+26,i+n+26,i,num);
		addEdge(i+n+26,dst,0,INF);
	}
	int cnt=0;
	while(spfa(0,dst))
		cnt+=modify(dst);
	if(cnt<need)
		cout<<-1<<endl;
	else
		cout<<ans<<endl;
	return 0;
}
